CONSTANT upper, $0FFF

LOAD.C   #SVF_F, $0020
LOAD.C   #SVF_Q, $0100

&start
LOAD.C   #SVF_IN, #upper
MOVE     #R_AUDIO, #SVF_OUT
LOAD.C   #R_ST1, 62000

&pos1
BNCC     #SR_ST1, #pos1
NOP
LOAD.C   #SVF_IN, 0
MOVE     #R_AUDIO, #SVF_OUT
LOAD.C   #R_ST1, 62000

&pos2
BNCC     #SR_ST1, #pos2
NOP
BRA      #start
